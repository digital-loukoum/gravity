### React example

```tsx
//  /components/Add.tsx
import { api } from "/api"
import { useState } from "react";

export function Add() {
  const [result, setResult] = useState<undefined | number>(undefined)

  api.math.add(1, 2).then(setResult)

  return (
    <>
      {result === undefined ? `Loading...` : result}
    </>
  )
}
```
